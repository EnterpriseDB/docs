---
title: "DBMS_XMLDOM"
---

The `DBMS_XMLDOM` package is mainly used to access the `XMLType` objects. It provides an application programming interface for HTML and XML documents as an Document Object Model (DOM). This package is specially designed to work like any other DOM interfaces.

The `DBMS_XMLDOM` package implements the following types:

| Type name     | Description                               |
|---------------|-------------------------------------------|
| `DOMNode`     | It implements the DOM Node interface.     |
| `DOMNodeList` | It implements the DOM NodeList interface. |
| `DOMElement`  | It implements the DOM Element interface.  |
| `DOMText`     | It implements the DOM Text interface.     |
| `DOMDocument` | It implements the DOM Document interface. |

This table lists the subprograms available with `DBMS_XMLDOM` package:

| Subprogram       | Type      | Return type | Description |
|------------------|-----------|-------------|-------------|
| `newDOMDcoument` | Function  |             |

| `createTextNode` | Function  |             |
| `setVersion`     | Procedure |             |
| `makeNode`       | Function  |             |
| `getxmltype`     | Function  |             |
| `freeDocument`   | Procedure |             |
| `makenode`       | Function  |             |
| `setAttribute`   | Procedure |             |
| `getAttribute`   | Function  |             |
| `makeNode`       | Function  |             |

CREATE OR REPLACE PACKAGE dbms_xmldom AUTHID CURRENT_USER AS

	--
	-- DOM interface types
	--
	TYPE DOMNode IS RECORD (id RAW(13)) OPTIONS (type_oid = "1054", type_rel_oid = "1055", type_array_oid = "1056");
	TYPE DOMNodeList IS RECORD (id RAW(13)) OPTIONS (type_oid = "1057", type_rel_oid = "1058", type_array_oid = "1059");
	TYPE DOMElement IS RECORD (id RAW(13)) OPTIONS (type_oid = "1211", type_rel_oid = "1212", type_array_oid = "1260");
	TYPE DOMText IS RECORD (id RAW(13)) OPTIONS (type_oid = "1234", type_rel_oid = "1235", type_array_oid = "1261");
	TYPE DOMDocument IS RECORD (id RAW(13)) OPTIONS (type_oid = "1629", type_rel_oid = "1630", type_array_oid = "2326");

	FUNCTION newDOMDocument RETURN DOMDocument;
	FUNCTION newDOMDocument(xmldoc IN xmltype) RETURN DOMDocument;
	FUNCTION newDOMDocument(cl IN clob) RETURN domdocument;
	FUNCTION createElement(doc DOMDocument, tagName IN VARCHAR2) RETURN DOMElement;
	FUNCTION createTextNode(doc DOMDocument, data IN VARCHAR2) RETURN DOMText;
	PROCEDURE setVersion(doc DOMDocument, v VARCHAR2);
	FUNCTION makeNode(doc DOMDocument) RETURN DOMNode;
	FUNCTION getxmltype(doc in DOMDocument) RETURN xmltype;
	PROCEDURE freeDocument(doc IN DOMDocument);

	FUNCTION makeNode(elem DOMElement) RETURN domnode;
	PROCEDURE setAttribute(elem DOMElement, name IN VARCHAR2, newvalue IN VARCHAR2);
	FUNCTION getAttribute(elem DOMElement, name IN VARCHAR2) RETURN VARCHAR2;
	FUNCTION getAttribute(elem DOMElement, name IN VARCHAR2,
						  ns IN VARCHAR2) RETURN VARCHAR2;

	FUNCTION makeNode(t DOMText) RETURN DOMNode;

	FUNCTION getNodeName(n DOMNode) RETURN VARCHAR2;
	FUNCTION getNodeValue(n domnode) RETURN VARCHAR2;
	FUNCTION getFirstChild(n DOMNode) RETURN DOMNode;
	FUNCTION getChildNodes(n DOMNode) RETURN DOMNodeList;
	FUNCTION appendChild(n DOMNode, newChild IN DOMNode) RETURN DOMNode;

	FUNCTION getLength(nl DOMNodeList) RETURN pls_INTEGER;
	FUNCTION item(nl DOMNodeList, idx IN PLS_INTEGER) RETURN DOMNode;

	-- Functions to cast DOMNode objects
	FUNCTION makeElement(n DOMNode) RETURN DOMElement;
END;
